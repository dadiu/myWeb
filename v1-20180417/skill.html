控制渐变/控制速度/控制右旋转

控制渐变
background-color:#333;
background-image:linear-gradient(to bottom, #333, #444);
background-image:-moz-linear-gradient(top, #333, #444); /* Firefox */
background-image:-webkit-gradient(linear, 0 0, 0 100%, from(#333), to(#444)); /* Safari 和 Chrome */
background-image:-webkit-linear-gradient(top, #333, #444); /* Safari 和 Chrome */
background-image:-o-linear-gradient(top, #333, #444); /* Opera */
background-repeat:repeat-x;
filter:progid:DXImageTransform.Microsoft.gradient(startColorstr='#ff333333', endColorstr='#ff444444', GradientType=0);/* ie7/8  */

控制速度
/* transition */
transition:all 0.2s ease-out; 
-ms-transition:all 0.2s ease-out; /* Internet Explorer */
-moz-transition:all 0.2s ease-out; /* Firefox */
-webkit-transition:all 0.2s ease-out; /* Safari 和 Chrome */
-o-transition:all 0.2s ease-out; /* Opera */

控制右旋转
/* Rotate div */
transform:rotate(90deg);
-ms-transform:rotate(90deg); /* Internet Explorer */
-moz-transform:rotate(90deg); /* Firefox */
-webkit-transform:rotate(90deg); /* Safari 和 Chrome */
-o-transform:rotate(90deg); /* Opera */
