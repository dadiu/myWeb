去重复

=2014-0424-面试版-只遍历一次-推荐

function noRepeat(arr){
    var res = [], obj = {}, len = arr.length;
    for(var i = 0; i < len; i++){
        if(!obj[arr[i]]){
            obj[arr[i]] = true;
            res.push(arr[i]);
        }
    }
    return res;
}

var test = [2,2,5,5,66,7,22,4,22];
console.log(noRepeat(test));
            
2014-0423-狮虎版-遍历次数比较多-推荐

var noRepeat = {
    inArry : function(obj , str){
        for(var key in str){
            if(str[key] === obj ){
                return true;
            };
        };
        return false;    
    },

    unique : function(str){
        var res = [];
        for(var key in str){
            if(!noRepeat.inArry(str[key], res)){
                res.push(str[key]);
            };
        };
        return res;
    }

},

test = [01,02,03,04,04,03,02,01,05,07,05];

console.log(noRepeat.unique(test));

数组删除空元素

                
function arrEmpty(arr){
    for(var i = 0, len = arr.length; i < len; i++ ){
      if(arr[i] == ''){
        arr.splice(i,1);
        i--;        //删除之后数i移出
        len--;      //删除之后长度变短
      }
    }
    return arr;
}
var arr = ['a','','b','c','d','',''];
console.log(arrEmpty(arr));
