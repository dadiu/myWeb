实例代码：

css

.m_tab_title { padding:0; margin:0 0 -1px 0; list-style-type:none; overflow:hidden; }
.m_tab_title li { cursor:pointer; float:left; padding:5px 15px; margin-right:10px; border:1px solid #ccc; background:#fff; }
.m_tab_title li.crt { border-bottom-color:#fff; }
.m_tab_main { width:450px; height:200px; padding:10px; border:1px solid #ccc; background:#fff; }
.m_tab_item { display:none; height:100%; }
.m_tab_item.crt { display:block; }

js
$('.m_tab_title').on('click', 'li', function(){
	$(this).addClass('crt').siblings().removeClass('crt');
	var name = $(this).attr('data-name');
	$('div[data-name=' + name + ']').show().siblings().hide();
})
