$.each 编程练习

定义一个json对象，用于保存学生的相关资料

		对于遍历一个数组，用$.each()来处理，简直爽到了极点。例如：
		var JSONP = [
		            {'name':'num1','string':'str','age':'1'},
		            {'name':'num2','string':'num','age':'2'},
		            {'name':'num3','string':'obj','age':'3'},
		            {'name':'num4','string':'arr','age':'4'},
		            {'name':'num5','string':'ble','age':'5'}
		        ];
		$.each(JSONP,function(i,n)
		{
		console.log('索引:'+i,'对应值为：'+n.name + ',' + n.string + ',' + n.age);
		});

		参数i为遍历索引值，n为当前的遍历对象.
		n = JSONP[i];
		n.name = JSONP[i].name;

实例代码：

		(function(){
			var JSONP = [

	            {'name':'num4','string':'arr','age':'4'},

	        ];

	        $.each(JSONP,function(i,n){

	        		'' + n.age +
