//获取当前日期
function today (){

    var today = new Date(),
        year = today.getFullYear(),
        month = today.getMonth() + 1,
        day = today.getDate();

    if ( month < 10 ) {
        month = '0' + month;
    }

    if ( day < 10 ) {
        month = '0' + day;
    }

    return (year + '-' + month + '-' + day);
}
